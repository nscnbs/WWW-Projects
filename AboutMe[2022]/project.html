class MainActivity : AppCompatActivity() {

    private var r1 = 0
    private var r2 = 0
    private var score = 0


    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        score = 0
        roll()
    }

    fun roll(){
        findViewById(R.id.points).text = "Punkty: " + score
        r1 = Random.nextInt(10)
        r2 = Random.nextInt(10)
        findViewById(R.id.leftClick).text = "" + r1
        findViewById(R.id.rightClick).text = "" + r2
    }

    fun buttonLeft(view: View) {
        if (r1>=r2){
            score++
            Toast.makeText(this, "Dobrze!!!", Toast.LENGTH_SHORT).show()
        }
        else{
            score--
            Toast.makeText(this, "Źle!!!", Toast.LENGTH_SHORT).show()
        }
        roll()
    }
    fun buttonRight(view: View) {
        if (r1<=r2){
            score++
            Toast.makeText(this, "Dobrze!!!", Toast.LENGTH_SHORT).show()
        }
        else{
            score--
            Toast.makeText(this, "Źle!!!", Toast.LENGTH_SHORT).show()
        }
        roll()
    }
}
